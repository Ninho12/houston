# Ícones

Coleção de ícones do Houston.

### Lista de ícones

<Playground>
  {() => {
    const size = 'large';
    const icons = [
      {
        icon: <Icon.Add size={size} />,
        name: 'Add'
      }
    ];
    console.log(Icon.Module);
    return (
      <div className='list-icons'>
        {icons
          .sort((a, b) => a.name.localeCompare(b.name))
          .map((icon, index) => (
            <div key={`icon-${index}`} className='icon-preview'>
              {icon.icon}
              <div className='name' title={icon.name}>
                {icon.name}
              </div>
            </div>
          ))}
      </div>
    );
  }}
</Playground>

Checkbox Field

O componente de caixa de seleção (Checkbox) permite que o usuário selecione um ou mais itens de um determinado grupo.

Importação

import CheckboxField from '@eduzz/houston-ui/Forms/Checkbox';

Variações

Formulário

Caso utilize o componente de caixa de seleção sem estar dentro de um contexto de formulário, é possível referenciá-lo através da propriedade form.

import useForm from '@eduzz/houston-forms/useForm';
import CheckboxField from '@eduzz/houston-ui/Forms/Checkbox';


interface IModel {
  name: string;
}


const MyComponent = () => {
  const form = useForm<IModel>({
    onSubmit: () => {}
  });


  return (
    <CheckboxField
      name='example-form'
      label='Exemplo formulário'
      description='Checkbox fora de um FormContext'
      form={form}
    />
  );
};
